@article{kwon2017predicting,
    abstract = {This article reports two ERP studies that exploited the classifier system of Mandarin Chinese to investigate semantic prediction. In Mandarin, in certain contexts, a noun has to be preceded by a classifier, which has to match the noun in semantically-defined features. In both experiments, an N400 effect was elicited in response to a classifier that mismatched an up-coming predictable noun, relative to a matching classifier. Among the mismatching classifiers, the N400 effect was graded, being smaller for classifiers that were semantically related to the predicted word, relative to classifiers that were semantically unrelated to the predicted word. Given that the classifier occurred before the predicted word, this result shows that fine-grained semantic features of nouns can be pre-activated in advance of bottom-up input. The studies thus extend previous findings based on a more restricted range of highly grammaticalized features such as gender or animacy in Indo-European languages (Szewczyk & Schriefers, 2013; Van Berkum, Brown, Zwitserlood, Kooijman, & Hagoort, 2005; Wicha, Bates, Moreno, & Kutas, 2003).},
