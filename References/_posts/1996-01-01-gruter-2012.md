@article{gruter2012grammatical,
    abstract = {Mastery of grammatical gender is difficult to achieve in a second language (L2). This study investigates whether persistent difficulty with grammatical gender often observed in the speech of otherwise highly proficient L2 learners is best characterized as a production-specific performance problem, or as difficulty with the retrieval of gender information in real-time language use. In an experimental design that crossed production/comprehension and online/offline tasks, highly proficient L2 learners of Spanish performed at ceiling in offline comprehension, showed errors in elicited production, and exhibited weaker use of gender cues in online processing of familiar (though not novel) nouns than native speakers. These findings suggest that persistent difficulty with grammatical gender may not be limited to the realm of language production, but could affect both expressive and receptive use of language in real time. We propose that the observed differences in performance between native and non-native speakers lie at the level of lexical representation of grammatical gender and arise from fundamental differences in how infants and adults approach word learning. {\textcopyright} The Author(s) 2012.},
