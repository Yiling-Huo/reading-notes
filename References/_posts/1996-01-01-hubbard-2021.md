@article{hubbard2021dividing,
    abstract = {Although we often seem to successfully comprehend language in the face of distraction, few studies have examined the role of sustained attention in critical components of sentence processing, such as integrating information over a sentence and revising predictions when unexpected information is encountered. The current study investigated the impact of attention on sentence processing using a novel dual-task paradigm. Participants read weakly and strongly constraining sentences with expected or unexpected endings while also tracking the motion of dots in the background, and their EEG was recorded. Under full attention, the amplitude of the N400 component of the ERP, a measure of semantic access, was reduced (facilitated) in a graded fashion by contextual strength and fit. This context-based facilitation was attenuated when attention was divided, suggesting that sustained attention is important for building up message-level representations. In contrast, the post-N400 frontal positivity that has been observed to prediction violations and associated with revision processes was unaffected by dividing attention. However, under divided attention, participants also elicited posteriorly-distributed effects to these violations. Thus, predictive processes seem to be engaged even when attention is divided, but additional resources may then be required to process unexpected information.},
