@article{gareze2007absence,
    abstract = {This chapter concentrates on the extraction of semantic information from extrafoveal vision. To investigate semantic processing, the semantic relationship between an object and the scene background in which it is located can be manipulated. An object is semantically related to the scene context when its identity is compatible with the scene's "gist." Research into scene context effects has claimed that the semantic relationship between a scene and an object can affect initial visual processing. It contributes to this debate, considering different scene types and methodologies. Using simple line drawings, complex naturalistic photographs and line drawings derived from them, the detectability of semantically consistent and inconsistent objects in scenes was investigated. Reliable evidence of a consistency effect on scene processing within a single fixation or on subsequent eye movements prior to target fixation were failed to found.},
