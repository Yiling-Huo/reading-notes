@incollection{satub2015reading,
    abstract = {Understanding what we read requires constructing a representation of each sentence’s syntactic structure and meaning. We are generally not consciously aware of the complex cognitive operations that underlie these processes. However, researchers have been able to learn a great deal about these operations by using methods such as eyetracking during reading and recording of Event-Related Potentials (ERPs). This chapter provides a broad overview of some of the main theoretical issues and empirical conclusions in this area, focusing on the following questions: How rapidly do we perform syntactic and semantic analysis in the course of reading? Do we construct more than one syntactic analysis at a time? Does syntactic analysis always precede semantic interpretation? What role does memory play in sentence processing? Finally, how do these sentence-level processes relate to the process of visual word recognition itself?},
