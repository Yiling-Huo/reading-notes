@article{cooper1975control,
    abstract = {In a current research program which may potentially be applied to investigate a wide range of topics in psychology, education, and communications, subjects' spontaneous eye movements are accurately monitored while they view visual displays (consisting of N multiplied by N arrays (N equals 2,3,4,5) of pictures of familiar objects and/or their printed names) and simultaneously listen to prerecorded spoken word lists, prose passages, or questionnaires. This is done using a dual Purkinje image eye tracker. All eye movement and speech event data are acquired on digital magnetic tape with a minicomputer for later off-line computer analysis. The same information is also available on video tape with sound track. The control of eye fixation by the meaning of spoken language could be applied to further the development of computer-assisted instruction by shortening the time to feedback and increasing the accuracy with which feedback is provided.},
