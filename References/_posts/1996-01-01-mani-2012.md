@article{mani2012prediction,
    abstract = {Are there individual differences in children's prediction of upcoming linguistic input and what do these differences reflect? Using a variant of the preferential looking paradigm (Golinkoff, Hirsh-Pasek, Cauley, & Gordon, 1987), we found that, upon hearing a sentence like, "The boy eats a big cake," 2-year-olds fixate edible objects in a visual scene (a cake) soon after they hear the semantically constraining verb eats and prior to hearing the word cake. Importantly, children's prediction skills were significantly correlated with their productive vocabulary size-skilled producers (i.e., children with large production vocabularies) showed evidence of predicting upcoming linguistic input, while low producers did not. Furthermore, we found that children's prediction ability is tied specifically to their production skills and not to their comprehension skills. Prediction is really a piece of cake, but only for skilled producers. {\textcopyright} 2012 American Psychological Association.},
