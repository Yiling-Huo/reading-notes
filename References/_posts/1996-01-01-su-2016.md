@article{su2016effect,
    abstract = {The processing of metaphors in the pattern 'A B' in Chinese was examined using the event related potentials (ERPs). In this study, ERPs were recorded when 24 participants read sentences with literal, metaphor or anomalous meanings towards the same critical word. The result shows that metaphor in 'A B' pattern can also evoke a larger N400 compared with the literal meaning. It suggests that the metaphoric interpretation was more difficult even with the similar familiarity in this pattern. Another finding suggests that the N400 effect mainly distributed in central and left hemisphere, which contradict to some of the previous studies. We suggest that both hemispheres may sensitive to metaphorical meaning but the left hemisphere was stimulated around 400ms. Further studies are needed to give more evidence to the hemisphere preference in Chinese metaphor compression.},
