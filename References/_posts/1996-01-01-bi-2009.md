@article{bi2009reading,
    abstract = {A recent hypothesis proposes that reading depends on writing in a logographic language - Chinese. We present a Chinese individual (HLD) with brain damage whose profile challenges this hypothesis. HLD was severely impaired in the whole process of writing. He could not access orthographic knowledge, had poor orthographic awareness, and was poor at delayed- and direct-copying tasks. Nevertheless, he was perfect at visual word-picture matching and read aloud tasks, indicating his intact ability to access both the semantics and phonology in reading. He was also able to distinguish between fine visual features of characters. We conclude that reading does not depend on writing, even in Chinese. {\textcopyright} 2008 Elsevier Ltd.
