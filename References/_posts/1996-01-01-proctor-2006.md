@article{proctor2006polarity,
    abstract = {Differences in performance with various stimulus-response mappings are among the most prevalent findings for binary choice reaction tasks. The authors show that perceptual or conceptual similarity is not necessary to obtain mapping effects; a type of structural similarity is sufficient. Specifically, stimulus and response alternatives are coded as positive and negative polarity along several dimensions, and polarity correspondence is sufficient to produce mapping effects. The authors make the case for this polarity correspondence principle using the literature on word-picture verification and then provide evidence that polarity correspondence is a determinant of mapping effects in orthogonal stimulus-response compatibility, numerical judgment, and implicit association tasks. The authors conclude by discussing implications of this principle for interpretation of results from binary choice tasks and future model development.
