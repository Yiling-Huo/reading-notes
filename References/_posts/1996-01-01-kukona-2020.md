@article{kukona2020lexical,
    abstract = {Two visual world experiments investigated the priming of form (e.g., phonology) during language processing. In Experiment 1, participants heard high cloze probability sentences like “In order to have a closer look, the dentist asked the man to open his{\ldots}” while viewing visual arrays with objects like a predictable target mouth, phonological competitor mouse, and unrelated distractors. In Experiment 2, participants heard target-associated nouns like “dentist” that were isolated from the sentences in Experiment 1 while viewing the same visual arrays. In both experiments, participants fixated the target (e.g., mouth) most, but also fixated the phonological competitor (e.g., mouse) more than unrelated distractors. Taken together, these results are interpreted as supporting association-based mechanisms in prediction, such that activation spreads across both semantics and form within the mental lexicon (e.g., dentist-mouth-mouse) and likewise primes (i.e., preactivates) the form of upcoming words during sentence processing.
